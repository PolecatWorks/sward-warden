25    let is_admin = crate::webserver::auth::check_is_admin(&state.db_pool, user_id).await;

27    let has_reports_module = sqlx::query_scalar::<_, bool>(

43            "SELECT id, user_id, name, location, has_derogation, photo, updated_at, is_deleted FROM farms WHERE user_id = $1 AND updated_at > $2"

10pub async fn list_events(

24pub async fn create_event(
25    State(state): State<AppState>,
26    UserId(user_id): UserId,
27    Json(event): Json<Event>,
28) -> Result<Json<Event>, AppError> {
29    // Verify field ownership
30    let field_belongs = sqlx::query_scalar::<_, bool>(
31        "SELECT EXISTS(SELECT 1 FROM fields f JOIN farms fa ON f.farm_id = fa.id WHERE f.id = $1 AND fa.user_id = $2 AND f.is_deleted = FALSE)"
32    )
33    .bind(event.field_id)
34    .bind(user_id)
35    .fetch_one(&state.db_pool)
36    .await?;
37
38    if !field_belongs {
39        return Err(AppError::Forbidden(
40            "Field is invalid or unauthorized".to_string(),
41        ));
42    }
43
44    let new_event = sqlx::query_as::<_, Event>(
45        "INSERT INTO events (field_id, event_type, description, date, mapp_number, eppo_code, bbch_growth_stage) VALUES ($1, $2, $3, $4, $5, $6, $7) RETURNING id, field_id, event_type, description, date, updated_at, is_deleted, mapp_number, eppo_code, bbch_growth_stage"
46    )
47    .bind(event.field_id)
48    .bind(&event.event_type)
49    .bind(&event.description)
50    .bind(&event.date)
51    .bind(&event.mapp_number)
52    .bind(&event.eppo_code)
53    .bind(&event.bbch_growth_stage)
54    .fetch_one(&state.db_pool)
55    .await?;
56    Ok(Json(new_event))
57}

59// References more than 3 PRDs
60pub async fn delete_event(
61    State(state): State<AppState>,
62    UserId(user_id): UserId,
63    Path(id): Path<i64>,
64) -> Result<StatusCode, AppError> {
65    let is_admin = crate::webserver::auth::check_is_admin(&state.db_pool, user_id).await;
66
67    let result = if is_admin {
68        sqlx::query("UPDATE events SET is_deleted = TRUE, updated_at = NOW() WHERE id = $1")
69            .bind(id)
70            .execute(&state.db_pool)
71            .await?
72    } else {
73        sqlx::query("UPDATE events SET is_deleted = TRUE, updated_at = NOW() WHERE id = $1 AND field_id IN (SELECT f.id FROM fields f JOIN farms fa ON f.farm_id = fa.id WHERE fa.user_id = $2)")
74            .bind(id)
75            .bind(user_id)
76            .execute(&state.db_pool)
77            .await?
78    };
79
80    if result.rows_affected() == 0 {
81        return Err(AppError::NotFound("Event not found".to_string()));
82    }
83
84    Ok(StatusCode::NO_CONTENT)
85}

1use crate::error::AppError;

4use tracing::info;

7pub async fn seed_database(pool: &PgPool, user_id: i64) -> Result<(), AppError> {
8    info!("Seeding database for user_id: {}", user_id);
9
10    // Ensure user exists
11    sqlx::query(
12        "INSERT INTO users (id, name, email, role) VALUES ($1, $2, $3, 'user') ON CONFLICT (id) DO NOTHING"
13    ).bind(user_id).bind("Demo User").bind(format!("user{}@example.com", user_id))
14    .execute(pool)
15    .await
16    .map_err(|e| AppError::Message(format!("Failed to ensure user: {e}")))?;
17
18    // Ensure admin user exists
19    sqlx::query(
20        "INSERT INTO users (id, name, email, role) VALUES ($1, $2, $3, 'admin') ON CONFLICT (id) DO NOTHING"
21    ).bind(999i64).bind("Demo Admin").bind("admin@example.com")
22    .execute(pool)
23    .await
24    .map_err(|e| AppError::Message(format!("Failed to ensure admin user: {e}")))?;
25
26    for i in 1..=3 {
27        let farm_name = format!("Farm {}", i);
28        let location = format!("County {}, NI", ["Down", "Antrim", "Tyrone"][i - 1]);
29
30        let farm_id: i64 = sqlx::query_scalar::<_, i64>(
31            "INSERT INTO farms (user_id, name, location, updated_at) VALUES ($1, $2, $3, $4) RETURNING id"
32        ).bind(user_id).bind(farm_name).bind(location).bind(Utc::now())
33        .fetch_one(pool)
34        .await
35        .map_err(|e| AppError::Message(format!("Failed to insert farm: {e}")))?;
36
37        for j in 1..=5 {
38            let field_name = format!("Field {}-{}", i, j);
39            let area = 2.5 + (j as f64) * 1.2;
40
41            let field_id: i64 = sqlx::query_scalar::<_, i64>(
42                "INSERT INTO fields (farm_id, name, area_hectares, updated_at) VALUES ($1, $2, $3, $4) RETURNING id"
43            ).bind(farm_id).bind(field_name).bind(area).bind(Utc::now())
44            .fetch_one(pool)
45            .await
46            .map_err(|e| AppError::Message(format!("Failed to insert field: {e}")))?;
47
48            let mut events_to_insert = Vec::new();
49
50            for k in 1..=10 {
51                let event_type = if k % 3 == 0 { "planned" } else { "completed" };
52                let description = format!("Slurry application #{} - {}m3 applied", k, k * 50);
53                let date = format!("2024-05-{:02}", k + 10);
54
55                events_to_insert.push((
56                    field_id,
57                    event_type.to_string(),
58                    description,
59                    date,
60                    Utc::now(),
61                ));
62            }
63
64            if !events_to_insert.is_empty() {
65                let mut query_builder = QueryBuilder::new(
66                    "INSERT INTO events (field_id, event_type, description, date, updated_at) ",
67                );
68
69                query_builder.push_values(events_to_insert, |mut b, event| {
70                    b.push_bind(event.0)
71                        .push_bind(event.1)
72                        .push_bind(event.2)
73                        .push_bind(event.3)
74                        .push_bind(event.4);
75                });
76
77                let query = query_builder.build();
78
79                query
80                    .execute(pool)
81                    .await
82                    .map_err(|e| AppError::Message(format!("Failed to insert events: {e}")))?;
83            }
84        }
85    }
86
87    info!("Seeding complete!");
88    Ok(())
89}

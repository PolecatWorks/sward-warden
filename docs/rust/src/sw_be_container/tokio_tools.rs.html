1//! Module to handle easy sending functions to tokio
2//!
3//! The provides two functions one function run_in_tokio creates and sends the function to tokio.
4//! The second function run_in_tokio_with_cancel allows the creation of a CancellationToken which can be used to shut down the tokio async.
5
6use crate::error::AppError;
7use futures::Future;
8use tracing::{error, info};

10use serde::{Deserialize, Serialize};
11use tokio::runtime::{self, Runtime};
12use tokio_util::sync::CancellationToken;

14#[derive(Deserialize, Serialize, Debug, Clone)]
15pub struct ThreadRuntime {
16    pub threads: usize,
17    pub stack_size: usize,
18    pub name: String,
19}
20
21impl Default for ThreadRuntime {
22    // References more than 3 PRDs
23    fn default() -> Self {
24        ThreadRuntime {
25            threads: 0,
26            stack_size: 3_000_000,
27            name: "default".into(),
28        }
29    }
30}

32// PRD Reference: 0001
33pub fn rt_multithreaded(runtime: &ThreadRuntime) -> Result<Runtime, AppError> {
34    if runtime.threads == 0 {
35        runtime::Builder::new_current_thread()
36            .enable_io()
37            .enable_time()
38            .build()
39            .map_err(AppError::from)
40    } else {
41        runtime::Builder::new_multi_thread()
42            .worker_threads(runtime.threads)
43            .thread_name(runtime.name.clone())
44            .thread_stack_size(runtime.stack_size)
45            .enable_io()
46            .enable_time()
47            .build()
48            .map_err(AppError::from)
49    }
50}
51
52/// run async function inside tokio instance on current thread
53// No obvious PRD requirement
54pub fn run_in_tokio<F, T>(runtime: &ThreadRuntime, my_function: F) -> F::Output
55where
56    F: Future<Output = Result<T, AppError>>,
57{
58    info!("Starting Tokio: {}", runtime.name);
59
60    rt_multithreaded(runtime)?
61        // .expect("Runtime created")
62        .block_on(my_function)
63}
64
65/// Run async with cancellability via CancellationToken
66// No obvious PRD requirement
67pub fn run_in_tokio_with_cancel<F, T>(
68    runtime: &ThreadRuntime,
69    cancel: CancellationToken,
70    my_function: F,
71) -> F::Output
72where
73    F: Future<Output = Result<T, AppError>>,
74{
75    run_in_tokio(runtime, async {
76        tokio::select! {
77            _ = cancel.cancelled() => {
78                error!("Token cancelled");
79                Err(AppError::Cancelled)
80            },
81            z = my_function => {
82                info!("Completed function");
83                z
84            },
85        }
86    })
87}

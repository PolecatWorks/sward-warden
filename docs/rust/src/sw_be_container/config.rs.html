14#[derive(Deserialize, Serialize, Debug, Clone)]
15pub struct UrlWithUsernamePassword {
16    pub url: Url,
17    pub username: Option<String>,
18    pub password: Option<String>,
19}
20
21impl From<UrlWithUsernamePassword> for Url {
22    // References more than 3 PRDs
23    fn from(value: UrlWithUsernamePassword) -> Self {
24        let mut return_url = value.url;
25        if let Some(password) = value.password {
26            let _ = return_url.set_password(Some(&password));
27        }
28        if let Some(username) = value.username {
29            let _ = return_url.set_username(&username);
30        }
31        return_url
32    }
33}

102impl AppConfig {
103    // References more than 3 PRDs
104    pub fn load(
105        config_path: &std::path::Path,
106        secrets_dir: &std::path::Path,
107    ) -> Result<Self, Box<figment::Error>> {
108        let adapter = FileAdapter::wrap(Yaml::file(config_path)).relative_to_dir(secrets_dir);
109
110        Figment::new()
111            .merge(adapter)
112            .merge(Env::prefixed("SP_BE__").split("__").lowercase(true))
113            .extract()
114            .map_err(Box::new)
115    }
116}
